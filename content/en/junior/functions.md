## Як ви розумієте замикання? Що буде виведено в консолі в цьому випадку?

```javascript
var f = function() {
  console.log(1);
}

var execute = function(f) {
  setTimeout(f, 1000);
}

execute(f); // что выведет в консоль и почему

f = function() {
  console.log(2);
}
```
